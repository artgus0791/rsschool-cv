class ZenBlogAdminContactController extends Controller
{
    private $contactService;

    public function __construct(ContactService $contactService)
    {
        $this->contactService = $contactService;
    }

    public function contacts(Request $request)
    {
        $contacts = $this->contactService->getSearchContacts($request->search);

        Contact::unread()->update(['is_read' => 1]);

        return view('zenblog.admin.contacts.index', compact('contacts'));
    }

    public function logout()
    {
        Auth::logout();

        return redirect()->route('zenblog.index');
    }

    public function editContact(int $id)
    {
        $contact = Contact::find($id);

        if (!$contact) {
            return redirect()->route('zenblog.admin.contact')->with('message', __('auth.message_not_found'));
        }
        return view('zenblog.admin.contacts.edit', compact('contact', ));
    }
    public function updateContact(Request $request, int $id)
    {
        $contact = Contact::find($id);

        if (!$contact) {
            return redirect()->route('zenblog.admin.contacts')->with('message', 'Сообщение  не найдено');
        }

        $contact->update(
            [
                'name' => $request->name,
                'email' => $request->email,
                'phone' => $request->phone,
                'subject' => $request->subject,
                'message' => $request->message,
                'is_read' => $request->is_read ? 1 : 0,
                'image' => $request->hasFile('image') ?
                    $this->contactService->updateImage($contact->image, $request->image) :
                    $contact->image
            ]
        );

        return redirect()->route('zenblog.admin.contacts.edit', [$contact->id]);
    }

    public function deleteAllContacts()
    {
        Contact::query()->delete();

        return redirect()->route('zenblog.admin.contacts')->with('success', 'Данные успешно удалены');
    }


}
